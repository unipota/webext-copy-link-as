function copyToClipboard(html, isHtml) {
  function listener(e) {
    e.clipboardData.setData(isHtml ? "text/html" : "text/plain", html);
    e.preventDefault();
  }
  document.addEventListener("copy", listener);
  document.execCommand("copy");
  document.removeEventListener("copy", listener);
}
